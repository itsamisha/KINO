CREATE TABLE Product (
  product_id SERIAL PRIMARY KEY,
  name VARCHAR(255) NOT NULL,
  price DECIMAL(10, 2) NOT NULL,
  stock_quantity INT NOT NULL,
  description TEXT,
  purchase_count INT,
  user_id INT REFERENCES users(user_id) ON DELETE CASCADE
);
